    # axios 是什么

    易用、简洁且高效的基于 promise 的 http 库
    官网 (opens new window)

    # 特性

    1. 支持 Node 和浏览器端

    2. 支持 Promise

    # 如何使用

    1. 安装
    npm i axios
    1
    1. 引入
    import axios from 'axios'

// get 请求
axios.get('user', {
    params: {
      ID: 12345
    }
  })
  .then(function (response) {
    console.log(response)
  })
  .catch(function (error) {
    console.log(error)
  })

// post 请求
axios.post('/user', {
    firstName: 'Fred',
    lastName: 'Flintstone'
  })
  .then(function (response) {
    console.log(response)
  })
  .catch(function (error) {
    console.log(error)
  })

// 并发的多个请求
function getUserAccount() {
  return axios.get('/user/12345')
}

function getUserPermissions() {
  return axios.get('/user/12345/permissions');
}

axios.all([getUserAccount(), getUserPermissions()])
    .then(axios.spread(function (data1, data2) { // 当两个请求都完成后执行
    console.log(data1, data2) // data1 —— 第一个请求的数据； data2 —— 第二个请求的数据
    }))

    # 配置默认

    全局的 axios 默认

    // 配置了默认的域名后，在发送请求时就不再需要带域名
axios.defaults.baseURL = 'https://api.example.com'

// 全局配置 post 请求体的格式
axios.defaults.headers.post['Content-Type'] = 'application/x-www-form-urlencoded'

    # 拦截器(在请求开始和拿到响应时)

    1. 可以用来做请求的进度条效果
    // 添加请求拦截器
axios.interceptors.request.use(function (config) {
    // 在发送请求之前做些什么
    return config;
  }, function (error) {
    // 对请求错误做些什么
    return Promise.reject(error);
  });

// 添加响应拦截器
axios.interceptors.response.use(function (response) {
    // 对响应数据做点什么
    return response;
  }, function (error) {
    // 对响应错误做点什么
    return Promise.reject(error);
  });

    1. 可以在客户端模拟获取数据
    let book = {
  name: '高级程序设计',
  number: 2,
  id: 1
}
axios.interceptors.response.use(function (response) {
  // es6 写法
  // 等价于 let { config } = response  let { method, url, data } = config
  let { config: { method, url, data }} = response // 这里的 data 是请求参数
  if(url === '/books/1' && method === 'get') {
    response.data = book
  } 
  return response
})

axios.get('/books/1')
    .then(({ data }) => { // es6 写法
    console.log(data)
